580  template <int dim> double IFData<dim>::BrentRootFinder(const RvDim & a_loPt,
581  const RvDim & a_hiPt,
582  const int & a_edgeDir) const
583  // const double & a_smallestRoot,
584  // const double & a_biggestRoot) const
585  {
586  PR_TIME("ifdata::brentrootfinder");
587  // Max allowed iterations and floating point precision
588  const unsigned int MAXITER = 100;
589  const double EPS = 3.0e-15;
590 
591  unsigned int i;
592  double aPt;
593  double bPt;
594  double c, fa, fb, fc;
595  double d, e;
596  double tol1, xm;
597  double p, q, r, s;
598 
599  aPt = -1.0; // a_smallestRoot;
600  bPt = 1.0; // a_biggestRoot;
601 
602  RvDim physCoordAPt = a_loPt;
603  RvDim physCoordBPt = a_hiPt;
604 
605  fa = -m_function->value(IndexTM<int,dim>::Zero,physCoordAPt);
606  fb = -m_function->value(IndexTM<int,dim>::Zero,physCoordBPt);
607 
608  // Init these to be safe
609  c = d = e = 0.0;
610 
611  if (fb*fa > 0)
612  {
613  std::cout << "fa " << fa << " fb " << fb << std::endl;
614  PR_error("IFData::BrentRootFinder. Root must be bracketed, but instead the supplied end points have the same sign.");
615  }
616 
617  fc = fb;
618 
619  for (i = 0; i < MAXITER; i++)
620  {
621  if (fb*fc > 0)
622  {
623  // Rename a, b, c and adjust bounding interval d
624  c = aPt;
625  fc = fa;
626  d = bPt - aPt;
627  e = d;
628  }
629 
630  if (std::abs(fc) < std::abs(fb))
631  {
632  aPt = bPt;
633  bPt = c;
634  c = aPt;
635  fa = fb;
636  fb = fc;
637  fc = fa;
638  }
639 
640  // Convergence check
641  tol1 = 2.0 * EPS * std::abs(bPt) + 0.5 * TOLERANCE;
642  xm = 0.5 * (c - bPt);
643 
644  if (std::abs(xm) <= tol1 || fb == 0.0)
645  {
646  break;
647  }
648 
649  if (std::abs(e) >= tol1 && std::abs(fa) > std::abs(fb))
650  {
651  // Attempt inverse quadratic interpolation
652  s = fb / fa;
653  if (aPt == c)
654  {
655  p = 2.0 * xm * s;
656  q = 1.0 - s;
657  }
658  else
659  {
660  q = fa / fc;
661  r = fb / fc;
662  p = s * (2.0 * xm * q * (q-r) - (bPt-aPt) * (r-1.0));
663  q = (q-1.0) * (r-1.0) * (s-1.0);
664  }
665 
666  // Check whether in bounds
667  if (p > 0) q = -q;
668 
669  p = std::abs(p);
670 
671  if (2.0 * p < std::min(((double)3.0)*xm*q-std::abs(tol1*q), std::abs(e*q)))
672  {
673  // Accept interpolation
674  e = d;
675  d = p / q;
676  }
677  else
678  {
679  // Interpolation failed, use bisection
680  d = xm;
681  e = d;
682  }
683  }
684  else
685  {
686  // Bounds decreasing too slowly, use bisection
687  d = xm;
688  e = d;
689  }
690 
691  // Move last best guess to a
692  aPt = bPt;
693  fa = fb;
694 
695  // Evaluate new trial root
696  if (std::abs(d) > tol1)
697  {
698  bPt = bPt + d;
699  }
700  else
701  {
702  if (xm < 0) bPt = bPt - tol1;
703  else bPt = bPt + tol1;
704  }
705 
706  physCoordBPt[a_edgeDir] = ((1.0 - bPt)/2.0) * a_loPt[a_edgeDir] + ((1.0 + bPt)/2.0) * a_hiPt[a_edgeDir];
707  fb = -m_function->value(IndexTM<int,dim>::Zero,physCoordBPt);
708  }
709 
710  if (i >= MAXITER)
711  {
712  cerr << "BrentRootFinder: exceeding maximum iterations: "
713  << MAXITER
714  << "\n";
715  }
716 
717  return bPt;
718  }

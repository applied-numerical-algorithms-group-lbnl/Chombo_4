152  void addLDeriv(
153  ScalarPowerMap& a_lDerivs,
154  ScalarPowerMap& a_phiDerivs,
155  NormalDerivativeMap& a_nDerivs,
156  const IvDim& a_deriv)
157  {
158  // If this is the first call, just calc l = |\grad phi|
159  if (a_deriv == IvDim::Zero)
160  {
161  RvDim dphi;
162  double len = 0;
163  for (int d=0; d < dim; ++d)
164  {
165  double val = a_phiDerivs[BASISV_TM<int,dim>(d)];
166  dphi[d] = val;
167  len += val*val;
168  }
169  a_lDerivs[a_deriv] = sqrt(len);
170  return;
171  }
172 
173  // Otherwise, assume we are moving in increasing x, then y...
174  int dir;
175  for (dir=0; dir < dim-1; ++dir)
176  if (a_deriv[dir] > 0)
177  break;
178 
179  double dl = 0;
180  IvDim q = a_deriv - BASISV_TM<int,dim>(dir);
181  typename NormalDerivativeMap::const_iterator it;
182  for (it = a_nDerivs.begin(); it != a_nDerivs.end(); ++it)
183  {
184  IvDim r = it->first;
185  if (r <= q)
186  {
187  double coef = nChoosek(q,r);
188  RvDim dn = a_nDerivs[r];
189  for (int d=0; d < dim; ++d)
190  {
191  IvDim dphip = q - r + BASISV_TM<int,dim>(d) +
192  BASISV_TM<int,dim>(dir);
193  dl += coef * dn[d] * a_phiDerivs[dphip];
194  }
195  }
196  }
197  a_lDerivs[a_deriv] = dl;
198  }

201  void addNDeriv(
202  NormalDerivativeMap& a_nDerivs,
203  ScalarPowerMap& a_phiDerivs,
204  ScalarPowerMap& a_lDerivs,
205  const IvDim& a_deriv)
206  {
207  RvDim dn = RvDim::Zero;
208  typename NormalDerivativeMap::const_iterator it;
209  for (it = a_nDerivs.begin(); it != a_nDerivs.end(); ++it)
210  {
211  IvDim q = it->first;
212  if ((q <= a_deriv) && (q != a_deriv))
213  {
214  double coef = nChoosek(a_deriv,q);
215  dn += (coef * a_lDerivs[a_deriv-q]) * it->second;
216  }
217  }
218 
219  double l = a_lDerivs[IvDim::Zero];
220  RvDim dphi;
221  for (int d=0; d < dim; ++d)
222  dphi[d] = a_phiDerivs[a_deriv + BASISV_TM<int,dim>(d)];
223  a_nDerivs[a_deriv] = (dphi - dn) / l;
224  }

227  // Calculates the multi-index binomial coefficient,
228  // product of "n choose k" for each index
229  static int nChoosek(const IvDim& a_n, const IvDim& a_k)
230  {
231  int numer = 1;
232  int denom = 1;
233 
234  IvDim diff = a_n - a_k;
235  PR_assert(diff >= IvDim::Zero);
236  for (int d = 0; d < dim; ++d)
237  {
238  for (int i=diff[d]+1; i <= a_n[d]; ++i)
239  numer *= i;
240  for (int i=2; i <= a_k[d]; ++i)
241  denom *= i;
242  }
243 
244  return numer / denom;

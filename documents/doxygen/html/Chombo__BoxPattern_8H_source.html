110  BoxPattern(const Chombo4::DisjointBoxLayout & a_grids,
111  const IntVect & a_ghost,
112  bool a_printStuff = false)
113  {
114  CH_TIME("BoxPattern::exchange constructor");
115 
116  m_recvPattern.resize(CH4_SPMD::numProc());
117  m_sendPattern.resize(CH4_SPMD::numProc());
118  Chombo4::DataIterator dit = a_grids.dataIterator();
119  Chombo4::NeighborIterator nit(a_grids);
120  for (dit.begin(); dit.ok(); ++dit)
121  {
122  const Chombo4::Box& grid = a_grids[dit];
123  int myProcID = CH4_SPMD::procID();
124  Chombo4::Box gridGhost(grid);
125  gridGhost.grow(a_ghost);
126 
127  for (nit.begin(dit()); nit.ok(); ++nit)
128  {
129  Chombo4::Box neighbor = nit.box();
130  int neiProcID = a_grids.procID(nit());
131  Chombo4::Box neighborGhost= neighbor;
132  neighborGhost.grow(a_ghost);
133 
134  Chombo4::Box gridGhostInter(neighbor & gridGhost);
135  if (!gridGhostInter.isEmpty())
136  {
137  //his data copies to my ghost
138  Chombo4::Box srcValid = neighbor;
139  Chombo4::Box dstValid = grid;
140  box_interaction_t item(Chombo4::DataIndex(nit()), dit(), gridGhostInter, gridGhostInter, neiProcID, myProcID, srcValid, dstValid);
141  if (myProcID == neiProcID)
142  { // local move
143  m_localPattern.push_back(item);
144  }
145  else
146  {
147  m_recvPattern[neiProcID].push_back(item);
148  }
149  }
150  Chombo4::Box neighborGhostInter(grid & neighborGhost);
151  if (!neighborGhostInter.isEmpty())
152  {
153  //my data copies to his ghost
154  //local case will be taken care of when the data iterator gets to the neighbor box
155  if(myProcID != neiProcID)
156  {
157  Chombo4::Box dstValid = neighbor;
158  Chombo4::Box srcValid = grid;
159  box_interaction_t item(dit(), Chombo4::DataIndex(nit()), neighborGhostInter, neighborGhostInter, myProcID, neiProcID, srcValid, dstValid);
160  m_sendPattern[neiProcID].push_back(item);
161  }
162  }
163  }
164  }
165  sortAllPatterns();

169  {
170  printStuff(string("exchange"));
171  }

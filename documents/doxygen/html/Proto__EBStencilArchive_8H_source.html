383  Point retval;

385  // and "black= odd number of nonzeros (the rest).
386 #if DIM==2

391  else if(icolor == 1)

393  retval = Point::Ones();//(1,1)
394  }
395  else if(icolor == 2)
396  {
397  retval = Point::Zeros() + Point::Basis(1);//(0,1)
398  }
399  else if(icolor == 3)
400  {
401  retval = Point::Zeros() + Point::Basis(0);//(1,0)
402  }
403  else
404  {
405  PROTO_ASSERT(false, "color not found");
406  }
407 #elif DIM==3
408  if(icolor == 0)
409  {
410  retval = Point::Zeros();//(0,0,0)
411  }
412  else if(icolor == 1)
413  {

415  }
416  else if(icolor == 2)
417  {
418  retval = Point::Zeros() + Point::Basis(1) + Point::Basis(2);//(0,1,1)
419  }
420  else if(icolor == 3)
421  {
422  retval = Point::Zeros() + Point::Basis(0) + Point::Basis(2);//(1,0,1)
423  }
424  else if(icolor == 4)
425  {
426  retval = Point::Zeros() + Point::Basis(1);//(0,1,0)
427  }
428  else if(icolor == 5)
429  {
430  retval = Point::Zeros() + Point::Basis(0);//(1,0,0)
431  }
432  else if(icolor == 6)
433  {
434  retval = Point::Zeros() + Point::Basis(2);//(0,0,1)
435  }
436  else if(icolor == 7)
437  {
438  retval = Point::Ones();//(1,1,1)
439  }
440  else
441  {
442  PROTO_ASSERT(false, "color not found");

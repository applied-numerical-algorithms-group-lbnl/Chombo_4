241  template <int dim> double CutCellMoments<dim>::changeMomentCoordinates(PthMoment & a_refinedMomentMap,
242  const IndexTM<int,dim> & a_monomial,
243  const IndexTM<double,dim> & a_refinedCenterDelta)
244  {
245 // This function outputs the moment in the coordinate system where the
246 // origin is the current cell center, given the moment map of the refined
247 // moments computed in the coordinate system where the origin is the
248 // refined cell center it uses a Taylor development around 0 of the function
249 // (x+xc)^p1*(y+yc)^p2*...
250  double moment = 0.0;
251 
252 // Degree is the degree of the monomial
253  int degree = 0;
254 
255  for (int i = 0; i < dim; i++)
256  {
257  degree += a_monomial[i];
258  }
259 
260 // Loop over the possible values of r in the development of the Taylor series
261 // The Taylor series has a finite expansion as the function's derivatives
262 // are zero for r>degree
263  for (int r = 0; r <= degree; r++)
264  {
265  if (r >= 0)
266  {
267 // Generate all the possible monomials of degree r and add up the moment
268  IndexTM<int,dim> derivative;
269 
270  for (int idir = 0; idir < dim; ++idir)
271  {
272  derivative[idir] = 0;
273  }
274 
275  while (true)
276  {
277  for (int j = 1; j < dim-1; ++j)
278  {
279  int t = r;
280  for (int k = j+1; k < dim; ++k)
281  {
282  t -= derivative[k];
283  }
284 
285  if (derivative[j] > t)
286  {
287  derivative[j+1] += 1;
288  derivative[j ] = 0;
289  }
290  else
291  {
292  break;
293  }
294  }
295 
296  if (derivative[dim-1] > r)
297  {
298  break;
299  }
300 
301  derivative[0] = r;
302 
303  for (int j = 1; j < dim; ++j)
304  {
305  derivative[0] -= derivative[j];
306  }
307 
308 // Add up the relevant term of the refined moment map to the map
309  double coeff = 1.0;
310  double factorial = 1.0;
311 
312  for (int idir = 0; idir < dim; idir++)
313  {
314  for (int j = 0; j < derivative[idir]; j++)
315  {
316  coeff *= a_monomial[idir] - j;
317  factorial *= j+1;
318  }
319 
320 // This is to prevent the computation of 0*0^-1 which result
321 // in a nan
322  if (a_monomial[idir]-derivative[idir] < 0)
323  {
324  if (coeff != 0)
325  {
326  PR_error("Coeff should be zero when the derivative has higher coefficient than the monomial");
327  }
328  }
329  else
330  {
331  coeff *= pow(a_refinedCenterDelta[idir],a_monomial[idir]-derivative[idir]);
332  }
333  }
334 
335 //pout()<<a_refinedMomentMap[derivative]<<endl;
336  moment += coeff * a_refinedMomentMap[derivative] / factorial;
337 
338 // Increments to get to the next derivative
339  derivative[1] += 1;
340  }
341  }
342  }
343 
344  return moment;
345  }

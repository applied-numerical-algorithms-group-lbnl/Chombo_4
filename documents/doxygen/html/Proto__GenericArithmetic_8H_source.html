2 // For any class that has operator+=, defines a non-member operator+. Ditto
3 // operator-=, *=, /=.

23  * For example:
24  * class Foo : public GenericArithmeticable<double,Foo>

26  * public:
27  * Foo() : GenericArithmeticable<double,Foo>(this)
28  * {
29  * }
